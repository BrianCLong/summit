    tenant_id,
    COUNT(*) as total_suggestions,
    SUM(CASE WHEN status = 'approved' THEN 1 ELSE 0 END) as approved,

          COUNT(*)::numeric * 100, 2) as approval_rate

GROUP BY tenant_id;

GROUP BY feature_name
ORDER BY total_uses DESC;
